14def get_last_metrics_processed_date(): 

15 # We want to give time to the store to proccess all the metrics, 

16 # since the metrics are processed during the night 

17 # https://github.com/canonical-web-and-design/snapcraft.io/pull/616 

18 three_hours = relativedelta.relativedelta(hours=3) 

19 last_metrics_processed = datetime.datetime.utcnow() - three_hours 

20 

21 one_day = relativedelta.relativedelta(days=1) 

22 return last_metrics_processed.date() - one_day

30 :param snap_id The snap id

35 :returns A dictionary with the filters for the metrics API, by default 

36 returns also the 'weekly_installed_base_by_country'.

38 end = get_last_metrics_processed_date()

40 if metric_bucket == "d":

81def build_snap_installs_metrics_query(snaps, get_filter=get_filter): 

82 """Transforms an API response from the publisher metrics 

83 

84 :param snaps: dict containing snaps we want metrics for 

85 :param get_filter: function that builds a single filter payload 

86 

87 :returns: A dict containing a filter for each snap in snaps 

88 or empty if there are no snaps

93 end = get_last_metrics_processed_date()

110def transform_metrics(metrics, metrics_response, snaps): 

111 """Transforms an API response from the publisher metrics 

112 

113 :param metrics_response: The JSON response from the metrics API 

114 

115 :returns: A dictionary with the metric information 

116 """ 

117 for metric in metrics_response["metrics"]: 

118 if metric["status"] == "OK": 

119 snap_id = metric["snap_id"] 

120 

121 snap_name = None 

122 for snaps_name, snaps_id in snaps.items(): 

123 if snaps_id == snap_id: 

124 snap_name = snaps_name 

125 

126 metrics["snaps"].append( 

127 {"id": snap_id, "name": snap_name, "series": metric["series"]} 

128 ) 

129 metrics["buckets"] = metric["buckets"] 

130 

131 return metrics
